```yaml
resources:
  pipelines:
    pipeline_name:
      name: "[${bundle.target}] Pipeline Name"

      # Target catalog and schema
      catalog: ${var.catalog}
      target: ${var.schema}

      # Pipeline libraries
      libraries:
        - glob:
            include: ../src/pipelines/<pipeline_folder>/transformations/**
      
      root_path: ../src/pipelines/<pipeline_folder>

      serverless: true

      # Pipeline configuration
      configuration:
        source_catalog: ${var.source_catalog}
        source_schema: ${var.source_schema}

      continuous: false
      development: true
      photon: true

      channel: current

      permissions:
        - level: CAN_VIEW
          group_name: "users"
```
